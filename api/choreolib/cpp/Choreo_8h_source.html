35 static choreo::ProjectFile GetProjectFile() {
36 if (LAZY_PROJECT_FILE.has_value()) {
37 return LAZY_PROJECT_FILE.value();
38 }
39
40 std::vector<std::filesystem::path> matchingFiles;
41
42 try {
43 for (const auto& entry :
44 std::filesystem::directory_iterator(CHOREO_DIR)) {
45 if (std::filesystem::is_regular_file(entry) &&
46 entry.path().extension() == TRAJECTORY_FILE_EXTENSION) {
47 matchingFiles.push_back(entry.path());
48 }
49 }
50
51 if (matchingFiles.size() == 0) {
52 FRC_ReportError(frc::warn::Warning,
53 "Could not find file in deploy directory!");
54 } else if (matchingFiles.size() > 1) {
55 FRC_ReportError(frc::warn::Warning,
56 "Found multiple project files in deploy directory!");
57 }
58
59 std::error_code ec;
60 auto fileBuffer =
61 wpi::MemoryBuffer::GetFile(matchingFiles[0].string(), ec);
62 if (!fileBuffer || !ec) {
63 FRC_ReportError(frc::warn::Warning,
64 "Could not open choreo project file");
65 }
66
67 wpi::json json = wpi::json::parse(fileBuffer->GetCharBuffer());
68 std::string version = json["version"];
69 if (kSpecVersion != version) {
70 throw fmt::format(".chor project file: Wrong version {}. Expected {}",
71 version, kSpecVersion);
72 }
73 choreo::ProjectFile resultProjectFile;
74 choreo::from_json(json, resultProjectFile);
75 LAZY_PROJECT_FILE = resultProjectFile;
76 } catch (const std::filesystem::filesystem_error&) {
77 FRC_ReportError(frc::warn::Warning, "Error finding choreo directory!");
78 } catch (const wpi::json::exception&) {
79 FRC_ReportError(frc::warn::Warning, "Error parsing choreo project file!");
80 }
81 return LAZY_PROJECT_FILE.value();
82 }

94 template <choreo::TrajectorySample SampleType>
95 static std::optional<choreo::Trajectory<SampleType>> LoadTrajectory(
96 std::string_view trajectoryName) {
97 if (trajectoryName.ends_with(TRAJECTORY_FILE_EXTENSION)) {
98 trajectoryName = trajectoryName.substr(
99 0, trajectoryName.size() - TRAJECTORY_FILE_EXTENSION.size());
100 }
101
102 std::string trajectoryFileName = fmt::format(
103 "{}/{}{}", CHOREO_DIR, trajectoryName, TRAJECTORY_FILE_EXTENSION);
104
105 std::error_code ec;
106 auto fileBuffer = wpi::MemoryBuffer::GetFile(trajectoryFileName, ec);
107 if (!fileBuffer || !ec) {
108 FRC_ReportError(frc::warn::Warning, "Could not find trajectory file: {}",
109 trajectoryName);
110 return {};
111 }
112
113 try {
114 return LoadTrajectoryString<SampleType>(
115 std::string{fileBuffer->GetCharBuffer().data(), fileBuffer->size()},
116 trajectoryName);
117 } catch (wpi::json::parse_error& ex) {
118 FRC_ReportError(frc::warn::Warning, "Could not parse trajectory file: {}",
119 trajectoryName);
120 FRC_ReportError(frc::warn::Warning, "{}", ex.what());
121 return {};
122 }
123 return {};
124 }

136 template <choreo::TrajectorySample SampleType>
137 static std::optional<choreo::Trajectory<SampleType>> LoadTrajectoryString(
138 std::string_view trajectoryJsonString, std::string_view trajectoryName) {
139 wpi::json json = wpi::json::parse(trajectoryJsonString);
140 std::string version = json["version"];
141 if (kSpecVersion != version) {
142 throw fmt::format("{}.traj: Wrong version {}. Expected {}",
143 trajectoryName, version, kSpecVersion);
144 }
145 choreo::Trajectory<SampleType> trajectory;
146 choreo::from_json(json, trajectory);
147 return trajectory;
148 }

165template <choreo::TrajectorySample SampleType>
166class TrajectoryCache {
167 public:
181 static std::optional<choreo::Trajectory<SampleType>> LoadTrajectory(
182 std::string_view trajectoryName) {
183 if (cache.contains(trajectoryName)) {
184 return cache[trajectoryName];
185 } else {
186 cache[trajectoryName] =
187 Choreo::LoadTrajectory<SampleType>(trajectoryName);
188 return cache[trajectoryName];
189 }
190 }
191
207 static std::optional<choreo::Trajectory<SampleType>> LoadTrajectory(
208 std::string_view trajectoryName, int splitIndex) {
209 std::string key = fmt::format("{}.:.{}", trajectoryName, splitIndex);
210
211 if (!cache.contains(key)) {
212 if (cache.contains(trajectoryName)) {
213 cache[key] = cache[trajectoryName].GetSplit(splitIndex);
214 } else {
215 auto possibleTrajectory = LoadTrajectory(trajectoryName);
216 cache[trajectoryName] = possibleTrajectory;
217
218 if (possibleTrajectory.has_value()) {
219 cache[key] = possibleTrajectory.value().GetSplit(splitIndex);
220 }
221 }
222 }
223
224 return cache[key];
225 }
226
230 static void Clear() { cache.clear(); }
231
232 private:
233 static inline std::unordered_map<std::string, choreo::Trajectory<SampleType>>
234 cache;
235};
