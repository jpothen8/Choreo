167 AutoTrajectory<SampleType> Trajectory(std::string_view trajectoryName,
168 AutoLoop<SampleType> loop) const {
169 std::optional<choreo::Trajectory<SampleType>> optTraj =
170 Choreo::LoadTrajectory<SampleType>(trajectoryName);
171 choreo::Trajectory<SampleType> trajectory;
172 if (optTraj.has_value()) {
173 trajectory = optTraj.value();
174 } else {
175 FRC_ReportError(frc::warn::Warning, "Could not load trajectory: {}",
176 trajectoryName);
177 }
178 AutoTrajectory<SampleType> autoTraj{
179 trajectoryName, trajectory, poseSupplier, controller,
180 mirrorTrajectory, trajectoryLogger, driveSubsystem, loop.GetLoop(),
181 autoBindings, loop.OnNewTrajectory};
182 loop.AddTrajectory(autoTraj);
183 return autoTraj;
184 }
